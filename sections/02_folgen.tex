\subsection{Definition}
\definition{Folge}: (unendliche) Liste von Zahlen, wobei jeder Eintrag ein \textbf{Glied der Folge} genannt wird.\\
Kann auch als Funktion/Abbildung $f:\mathbb{N}_0 \rightarrow \mathbb{R}$ (oder $f:\mathbb{N} \rightarrow \mathbb{R}$) aufgefasst werden.

\definition{Bilder} $a(n)=a_n$ sind Elemente der Folge.

% ------------------------------
\subsection{Darstellung}
\definition{direkt}: explizit angeben, wie ein beliebiges Folgenglied berechnet werden kann
$a_n = f(n)$

\definition{rekursiv}: Konstruktion eines Folgenglieds aus den vorangehenden
$a_n=g(a_{n-1}, a_{n_2},...)$

$f$ und $g$ nennt man \textbf{Bildungsgesetz}

% -----------------------------
\subsection{Eigenschaften}

Eine Folge $(a_n)_{n \in \mathbb{N}_0}$ ist \definition{nach unten/oben beschränkt} falls $M=\{a_N | n \in \mathbb{N}_0\}$ eine untere/obere Schranke besitzt.

Eine nach oben und nach unten beschränkte Folge ist \definition{beschränkt}.

Eine Folge ist \definition{(streng) monoton fallend} falls gilt $\forall n, m \in \mathbb{N}_0 : m > n \implies a_m \leq (<) a_n$

Eine Folge ist \definition{(streng) monoton wachsend} falls gilt $\forall n, m \in \mathbb{N}_0 : m > n \implies a_m \geq (>) a_n$

% ------------------------------
\subsection{Konvergenz / Divergenz}

Eine Folge ist \definition{konvergent} falls ein $L \in \mathbb{R}$ existiert, so dass $\forall \epsilon > 0 \exists N > 0$ so dass $\forall n > N : |a_n -L| < \epsilon$.

Eine konvergente Folge besitzt\\ \theorem{genau einen eindeutigen Grenzwert}.

$L$ ist der \definition{Grenzwert} $\lim_{n \rightarrow \infty} a_n = L$.

Falls kein $L$ existiert, nennt man die Folge \definition{divergent}.

$A \in \mathbb{R}$ ist \definition{Häufungspunkt} einer Folge $(a_n)_n\in\mathbb{N}_0$, falls $\forall \epsilon > 0 \forall N \in \mathbb{N}_0 \exists n \geq N$ so dass $|a_n - A | < \epsilon$.
\textit{Jedes Intervall der Form $(A- \epsilon, A+ \epsilon)$ enthält $a_n$ für unendlich viele $n$.}

$A\in \mathbb{R}$ ist Häufungspunkt von $(a_n)_n\in\mathbb{N}_0$ \lemma{$\Leftrightarrow$} es existiert eine konvergente Teilfolge mit $\lim_{k \rightarrow \infty} a_{n_k} = A$.

\greenBox{Konvergente Folge hat genau einen Häufungspunkt} (ist identisch mit Grenzwert).

\greenBox{Jede beschränkte und monotone Folge konvergiert.}

\lemma{Jede konvergente Folge ist beschränkt.}

\begin{enumerate}
    \item Eine monotone Folge reeller Zahlen $(a_n)_{n \in \mathbb{N}_0}$ konvergiert genau dann, wenn sie beschränkt ist.
    \item Falls $(a_n)_{n \in \mathbb{N}_0}$ monoton wachsend ist, gilt $\lim_{n \rightarrow \infty} a_n = \sup\{a_n | n \in \mathbb{N}_0\}$
    \item Falls $(a_n)_{n \in \mathbb{N}_0}$ monoton fallen ist, gilt $\lim_{n \rightarrow \infty} a_n = \inf\{a_n | n \in \mathbb{N}_0\}$
\end{enumerate}

\subsubsection{Spezialfälle divergenter Folgen}
$\lim_{n\rightarrow \infty} a_n = \infty$ falls $\forall M > 0 \exists N>0 \text{ so dass } \forall n>N : a_n > M$
\par
$\lim_{n\rightarrow \infty} a_n = -\infty$ falls $\forall M > 0 \exists N>0 \text{ so dass } \forall n>N : a_n <- M$

\subsubsection{Beispiele}
$a_n = x^n$: falls $|x|<1$ konvergent gegen null, falls $|x|>1$ divergent

$b_n = n ^{-p}$: konvergiert für $p>0$, divergiert für $p<0$

\subsubsection{Rechenregeln}
Falls $\lim_{n \rightarrow \infty} a_n = K ( \neq \pm \infty)$, $\lim_{n \rightarrow \infty} b_n = L ( \neq \pm \infty)$ und $C$ beliebige feste Zahl.

\subsubsection{Wichtige Grenzwerte}

\begin{enumerate}
    \item $lim_{n \rightarrow \infty} \frac{\log n}{n} = 0 = \lim_{n \rightarrow \infty} \frac{\ln n}{n}$
    \item $\lim_{n \rightarrow \infty} n^{1/n} = 1$
    \item $\lim_{n \rightarrow \infty} x^{1/n} = 1, x>0$
    \item $\forall x \in \mathbb{R}: \lim_{n \rightarrow \infty}(1+\frac{x}{n})^n = e^x$
    \item $\forall x \in \mathbb{R}: \lim_{n \rightarrow \infty}\frac{x^n}{n!} = 0$
\end{enumerate}

\theorem{Sandwich-Theorem} $(a_n)_{n\in\mathbb{N}_0}$ und $(c_n)_{n\in\mathbb{N}_0}$ mit $\lim_{n \rightarrow \infty} a_n = L$ und $\lim_{n \rightarrow \infty} c_n = L$ und $(b_n)_{n \in \mathbb{N}_0}$ mit der Eigenschaft,
dass $\exists N_0 \in \mathbb{N}$ mit $a_n \leq b_n \leq c_n \forall n \geq N_0$. $\implies$ $(b_n)_{n \in \mathbb{N}_0}$ ist konvergent und $\lim_{n \rightarrow \infty} b_n = L$

\subsection{Limes superior / inferior}
\definition{Limes superior} von $a_n$: $\limsup_{n \rightarrow \infty} a_n = \overline{\lim_{n \rightarrow \infty}}a_n=\lim_{n \rightarrow \infty} \overbrace{\sup\{a_k | k \geq n\}}^{=s_n}$
\textit{$(s_n)_{n\in \mathbb{N}_0}$ ist monoton fallend.}

\definition{Limes inferior} von $a_n$: $\liminf_{n \rightarrow \infty} a_n = \underline{\lim_{n \rightarrow \infty}}a_n=\lim_{n \rightarrow \infty} \inf\{a_k | k \geq n\}$

\greenBox{Häufungspunkt und limes superior} Sei $A=\limsup_{n \rightarrow \infty} a_n$. Dann ist $A$ Häufungspunkt und $\forall \epsilon > 0$ gilt:
\begin{enumerate}
    \item es nur unendlich viele $a_n$ gibt mit $a_n \geq A + \epsilon$
    \item für unendlich viele $a_n$ gilt $A- \epsilon < a_n < A + \epsilon$fk
\end{enumerate}

Limes superior (inferior) gibt den grössten (kleinsten) möglichen Häufungspunkt an.

\greenBox{Jede beschränkte Folge hat mindestens einen Häufungspunkt.} (reelle Zahlen)
\textit{Konvergenz hier über den gewohnten Abstand/Absolutbetrag definiert.}

\subsection{Teilfolgen}
\definition{Teilfolge} ist Folge $(a_{n_k})_{k\in \mathbb{N}_0}$, wobei $(n_k)_{k\in \mathbb{N}_0}$ Folge nicht-negativer ganzer Zahlen ist mit $\forall k\in \mathbb{N}_0: n_{k+1}>n_k$.

Für konvergente Folge $(a_n)_{n \in \mathbb{N}_0}$ \textbf{mit Grenzwert $L$} konvergiert jede Teilfolge \lemma{gegen den gleichen Grenzwert $L$}.

\subsection{Cauchy-Folgen}
$(a_n)_{n \in \mathbb{N}_0 \subset \mathbb{R}}$ heisst \definition{Cauchy-Folge}, falls $\forall \epsilon >0$ ein $N \in \mathbb{N}$ existiert, so dass gilt $\forall m, n \geq N |a_n - a_m| < \epsilon$.

\lemma{Jede Cauchy-Folge ist beschränkt.}

\definition{Reihe} Summe von Folgegliedern


\definition{Harmonische Summe} $\sum_{k=1}^n \frac{1}{k}$

$c_n=1 + \frac{1}{2} + \frac{1}{3} + ... + \frac{1}{n} = \sum_{k=1}^n \frac{1}{k}$ divergiert. Kann keine Cauchy-Folge sein.

\textit{(im Gegensatz zu $\sum_{k=1}^{n} \frac{1}{k^2}$)}

Eine Folge reeller Zahlen \theorem{konvergiert genau dann,} wenn sie eine \theorem{Cauchy-Folge ist}.
