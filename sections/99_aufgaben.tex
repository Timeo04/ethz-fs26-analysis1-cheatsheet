\subsection{Integrale}

\subsubsection{Integral mit Partialbruchzerlegung}
\textbf{Aufgabe:} $\int\frac{1}{x^{2}+x}dx$
\begin{itemize}
    \item \textbf{Ansatz:} $\frac{1}{x^2+x} = \frac{A}{x}+\frac{B}{x+1} \Rightarrow 1 = A(x+1) + Bx \Rightarrow A=1, B=-1$
    \item \textbf{Integrieren:} $\int(\frac{1}{x}-\frac{1}{x+1})dx = \ln(|x|)-\ln(|x+1|)+C = \ln\left(\left|\frac{x}{x+1}\right|\right)+C$
\end{itemize}

\subsubsection{Integral mit Substitution}
\textbf{Aufgabe:} $\int -\sin(x)\cos(x)dx$
\begin{itemize}
    \item \textbf{Ansatz \& Ableitung:} $u=\cos(x) \Rightarrow \frac{du}{dx}=-\sin(x)$
    \item \textbf{Einsetzen \& Integrieren:} $\int u \cdot \frac{du}{dx} dx = \int u du = \frac{1}{2}u^2+C = \frac{1}{2}\cos(x)^2+C$
\end{itemize}
