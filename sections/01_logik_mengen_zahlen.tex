\subsection{Logik}
\definition{Aussage} wohlformulierte mathematische Behauptung, die entweder wahr oder falsch ist.

\definition{Prädikat $A(x,y)$} Aussage, die von einer oder mehreren Variablen (Argumente, Inputs) abhängt.

\definition{Quantoren} $\forall, \exists, \exists!, \nexists$ \textit{(Reihenfolge wichtig)}

\greenBox{Satz} $\lnot (\forall x A(x)) \Leftrightarrow \exists x \lnot A(x)$, 
$\lnot (\exists x A(x)) \Leftrightarrow \forall x \lnot A(x)$,
$\forall x (A(x) \land B(x)) \Leftrightarrow (\forall x A(x)) \land (\forall x B(x))$,
$\exists x (A(x) \lor B(x)) \Leftrightarrow (\exists x A(x)) \lor (\exists x B(x))$,

\definition{Grundverknüpfungen} $\land, \lor, \implies (\lnot A \lor B), \Leftrightarrow$

$A \implies B$: A \definition{hinreichend}, B \definition{notwendig}

\lemma{$A \implies B \equiv \lnot B \implies \lnot A$}

\subsubsection{Max. / Min. / Inf. / Sup.}
\definition{Obere (untere) Schranke} $\forall x \in X : x \leq y (x \geq y)$

\definition{Maximum (Minimum)} $x_0 \in X$, $\forall x \in X : x \leq x_0 (x \geq x_0)$

eindeutig bestimmt (falls sie existieren)

\definition{Supremum (Infimum)} kleinste obere (grösste untere) Schranke

Jede nicht leere, nach oben (unten) beschränkte Teilmenge hat ein eindeutiges Supremum (Infimum).

$(\forall x \in X : x \leq S) \land (\forall \epsilon > 0 \exists x \in X : x>S-\epsilon)$

$\sup(\emptyset) = - \infty$ (Konvention)

\theorem{$\mathbb{R}$ ist geordneter, vollständiger Körper.}

\begin{enumerate}
    \item Additive und multiplikative Inverse eindeutig
    \item $\forall x \in \mathbb{R} : 0 * x = 0$
    \item $\forall x \in \mathbb{R} : (-1) * x = -x$
    \item $\forall y \in \mathbb{R} : y \geq 0 = -y \leq 0$
    \item $\forall y \in \mathbb{R} : y^2 \geq 0$
    \item Aus $x \leq y$ und $u \leq v$ folgt $x + u \leq y + v$
    \item Falls gilt $0 \leq x \leq y$ und $u \geq 0$, dann $xu \leq yu$
\end{enumerate}

\greenBox{\textbf{Archimedisches Prinzip}}
\begin{enumerate}
    \item Für $x \in \mathbb{R}$ und $y > 0$ existiert $n \in \mathbb{N}$ mit $ny > x$
    \item Für jedes $\epsilon > 0$ existiert $n \in \mathbb{N}$ mit $\frac{1}{n} < \epsilon$
\end{enumerate}

\subsubsection{Absolutbetrag}
$|x| := \max\{x, -x\}$

\begin{enumerate}
    \item $|x| \geq 0$
    \item $|x+y| \leq |x| + |y|$ (Dreiecksungleichung)
    \item $|x + y| \geq ||x|-|y||$
\end{enumerate}

\greenBox{Youngsche Ungleichung} Für jedes $\epsilon > 0$ und alle $x, y \in \mathbb{R}$ gilt: $2|xy| \leq \epsilon x^2 + \frac{1}{\epsilon}y^2$

\subsection{Vektorräume}

\definition{Skalarprodukt} $x \cdot y = <x, y> = [x,y] := \sum_{i=1}^{n}x_iy_i$

\definition{Euklidische Norm} $||x|| := \sqrt{\sum_{i=1}^{n} x_i^2}$

\definition{Euklidischer Abstand} $d(x,y):=\sqrt{\sum_{i=1}^{n}(x_i - y_i)^2}=||x-y||$

\lemma{Dreiecksungleichung} Für alle $x,y,z \in \mathbb{R}^n$: $||x-z||\leq ||x-y|| + ||y-z||$
