\subsection{Definition / Klassifizierung}

Eine \definition{Differentialgleichung} ist eine Gleichung, in der die gesuchte Funktion, sowie deren Ableitungen auftreten.

\begin{exercise}
    \textbf{Beispiele:} 

    \textit{$u'(t)=f(a-u(t)), u'(t)=r(a-u(t))(b-u(t)), u'(t)=ru(t)(1-\frac{u(t)}{L})$ wobei $r,a,b \text{ und }L$ Konstanten sind, oder auch $y^{(4)}(x)-4y^{(3)}(x)+5y^{(2)}(x)-4y'(x)+4y(x)=0$}    

    \textit{
    $t,x$ heissen \textbf{\definition{unabhĂ¤ngige} Variablen},
    $u,y$ heissen \textbf{\definition{abhĂ¤ngige} Variablen}
    }
\end{exercise}

Eine Differentialgleichung der Form 
$$a_n(x)y^{(n)}(x)+\dots+a_1(x)y'(x)+a_0(x)y(x)=s(x)$$ 
heisst \definition{lineare Differentialgleichung} mit \textbf{Koeffizienten $a_i(t)$}.

Die Funktion $s(x)$ nennen wir \definition{StĂ¶rfunktion.}

Falls gilt $s(x)=0$ heisst die Differentialgleichung \definition{homogen}, andernfalls \definition{inhomogen}.

Falls alle Funktionen $a_i(x)$ in der Differentialgleichung konstante Funktionen sind, nennen wir diese eine \definition{Differentialgleichung mit konstanten Koeffizienten}.

Die \definition{Ordnung} der Differentialgleichung ist die \textbf{maximale Ordnung der vorkommenden Ableitungen}.

\subsection{Lineare, homogene Differentialgleichungen}
$$a_n u^{(n)}(x) + a_{n-1} u^{(n-1)}(x) \dots + a_1 u'(x) + a_0 u(x) = 0$$

\theorem{Superpositionsprinzip}
Sind $y_1(x)$ und $y_2(x)$ LĂ¶sungen einer linearen, homogenen Differentialgleichung, so ist auch $y(x)=C_1y_1(x)+C_2y_2(x), C_1, C_2 \in \mathbb{R}$ eine LĂ¶sung derselben Differentialgleichung.

\textit{Geht nur fĂĽr homogene Differentialgleichungen (weil falls $C_1=C_2=0$)}

\theorem{FundamentallĂ¶sungen}
Eine lineare, homogene Differentialgleichung der Ordnung $n$ besitzt $n$ LĂ¶sungen $y_1(x), \dots, y_n(x)$, so ist die allgemeine LĂ¶sung dieser Gleichung gegeben durch $y(x)=C_1y_1(x)+\dots+C_ny_n(x), C_1, \dots, C_n \in \mathbb{R}$.
Diese LĂ¶sungen werden \definition{FundamentallĂ¶sungen} genannt.

\subsubsection{Allgemeine LĂ¶sung finden (mit konstanten Koeffizienten)}

\definition{Eulerscher Ansatz} $u(x)=e^{\lambda x}$

\definition{Charakteristische Gleichung} $a_n \lambda^n + a_{n-1} \lambda^{n-1}+ \dots + a_1 \lambda + a_0 =0$
\textit{(folgt durch Einsetzen des Eulerschen Ansatzes in Gleichung (und dann Ausklammern von $e^{\lambda x}$))} 

\definition{Charakteristisches Polynom} $p(\lambda) = a_n \lambda^n + a_{n-1} \lambda^{n-1}+ \dots + a_1 \lambda + a_0$.

\textit{
    Jede einfache relle Nullstelle $\lambda$ des charakteristischen Polynoms erzeugt die FundamentallĂ¶sung $u(x)=e^{\lambda x}$\\
    Jede Paar zweier zueinande komplex konjugierter Nullstellen $\lambda_{1,2}=a\pm ib$ erzeugt die beiden FundamentallĂ¶sungen $u_1(x)=e^{ax}\sin(bx)$, $u_2(x)=e^{ax}\cos(bx)$\\
    Jede Nullstelle $\lambda$ mit MultiplizitĂ¤t $s$ erzeugt die $s$ FundamentallĂ¶sungen $x^ke^{\lambda x}, k=0,...,s-1$
}

\greenBox{Allgemeine LĂ¶sung:}
\begin{itemize}
    \item Falls alle $r$ Nullstellen $\lambda_i$ des charakteristischen Polynoms reell sind und MultiplizitĂ¤t $m_i$ haben, gilt $u(x)=\sum_{i=0}^{r}(\sum_{p=0}^{m_i - 1}C_{ip} x^p e^{\lambda_i x})$
    \item Falls wir $r$ reelle Nullstellen und $s$ \textbf{Paare} komplexer Nullstellen $(a_j \pm ib_j)$ mit jeweils MultiplizitĂ¤t $m_i$ haben, gilt $u(x)=\sum_{i=0}^{r}(\sum_{p=0}^{m_i - 1}C_{ip} x^p e^{\lambda_i x}) + \sum_{j=0}^{s} (\sum_{q=0}^{m_j - 1}(A_{jq} x^q e^{a_j x} \sin(b_j x)+B_{jq} x^q e^{a_j x} \cos(b_j x)))$    
\end{itemize}

\begin{exercise}
    Allgemeine LĂ¶sung einer linearen, homogenen Differentialgleichung mit Parametern, bspw. $u''(x)-(a+b)u'(x)+abu(x)=0$

    \textit{Charakteristisches Polynom lĂ¶sen; ergibt $\lambda_1 = a, \lambda_2 = b$, dann Fallunterscheidung fĂĽr $a\neq b$ und $a=b$. Separat allgemeine LĂ¶sungen bestimmen (siehe oben).}

    \small{\textit{Wenn man eine Anfangswertaufgabe mit dieser Gleichung anschaut, kann man mithilfe von Taylorreihen und Vereinfachen zeigen, dass die LĂ¶sung fĂĽr $a \neq b$ und $b \rightarrow a$ gegen die LĂ¶sung fĂĽr den Fall $a=b$ konvergiert.}}
\end{exercise}

\subsection{Differentialgleichungen erster Ordnung}

\subsubsection{LĂ¶sung finden}

\greenBox{\textbf{Trennung der Variablen}}\\
\textit{"Alles was mit x zu tun hat auf eine Seite, alles andere auf die andere Seite"},
\textbf{eignet sich fĂĽr $y'(x)=f(x)\cdot g(y)$}.

\begin{enumerate}
    \item Differentialgleichung in Form $\frac{dy}{dx}=f(x)g(x)$ bringen
    \item Umstellen zu $\frac{1}{g(y)}dy=f(x)\,dx$
    \item Beidseitig integrieren und berechnen \textit{(mit Konstanten!)}
    \item Nach $y$ auflĂ¶sen
\end{enumerate}

\begin{exercise}
    \textbf{Beispiel: Newtonsches AbkĂĽhlungsgesetz}\\
    LĂ¶se $y'(t)=-k(y-y^*) \iff \frac{dy}{dt}=-k(y-y^*)=f(t) \cdot g(y)$ mit $f(t)=-k$ und $g(y)=y-y^*$

    $\implies \int \frac{1}{y-y^*}\,dy = \int -k\,dt$\\
    $\implies \ln|y-y^*|=-kt+C,C \in \mathbb{R}$\\
    $\implies |y-y^*|=e^{-kt+C}=e^{-kt}e^C=Ce^{-kt}, C \in \mathbb{R}$\\
    $\implies y-y^*=Ce^{-kt}, C \in \mathbb{R}$\\
    $\implies y = y(t) = y^* + Ce^{-kt}, C \in \mathbb{R}$

\end{exercise}

\greenBox{\textbf{Substitution}} (EinfĂĽhrung einer neuen Variable)\\
\textbf{Eignet sich fĂĽr $\frac{du}{dx}=u'(x)=f(ax+bu(x)+c)$} (Rechte Seite linear in $x$ und in $u$).

\begin{enumerate}
    \item Neue Variable/Funktion linear in $x$ und $u(x)$ einfĂĽhren ($z(x):=ax+bu(x)+c$)
    \item Ableiten
    \item $\frac{du}{dx}$ ersetzen durch $z$
    \item Mit Trennung der Variablen lĂ¶sen
    \item RĂĽcksubstituieren: $z(x)= ax+bu(x)+c = \dots$ umformen nach $u(x)$.
\end{enumerate}

\begin{exercise}
    \textbf{Beispiel:} $u'(x)=-3x+2u(x)-5$. 
    
    Neue Variable/Funktion: $z(x):=-3x+2u(x)-5$. 
    Dann gilt: $\frac{dz}{dx} = -3 + 2 \frac{du}{dx} = -3 + 2z$ ($\rightarrow$ trennbar).
    
    LĂ¶sen ergibt $z(x)=\frac{3}{2}+De^{2x}, D \in \mathbb{R}$, dann rĂĽcksubstituieren: $z(x)=-3x+2u(x)-5=\frac{3}{2}+De^{2x}$\\
    $\implies u(x)=De^{2x} + \frac{3}{2}x + \frac{13}{4}, D \in \mathbb{R}$
\end{exercise}

\textbf{Eignet sich auch fĂĽr $\frac{du}{dx}=u'(x)=f(\frac{y}{x})$} (\textbf{homogen in den Variablen}, \textit{"Rechte Seite hĂ¤ngt nur vom VerhĂ¤ltnis $\frac{x}{y}$ ab"}).

    \item $z(x)=\frac{y}{x} \iff z(x)\cdot x = y(x)$

    \item Umstellen nach $\frac{dz}{dx}= f(x)\cdot g(z)$ und Trennung der Variablen anwenden.
    \item RĂĽcksubstituieren: $z(x)=\dots=\frac{y(x)}{x}$ nach $y(x)$ auflĂ¶sen.
\end{enumerate}

\begin{exercise}
    \textbf{Beispiel:} $y'(x)=\frac{y}{x}+\frac{x^2}{y^2}$

    Neue Variable/Funktion: $z(x)=\frac{y}{x}\iff z(x)\cdot x=y(x)$
